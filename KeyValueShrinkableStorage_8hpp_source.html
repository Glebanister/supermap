212  [[nodiscard]] KeyIndexStorage shrink(IndexT shrinkBatchSize, const std::string &newIndexFileName) {
213 
214  const std::string shrinkFilenamePrefix = "shrink";
215  const std::string tempSortedIndexFilename = shrinkFilenamePrefix + "-sorted-keys";
216 
217  std::size_t batchesCount = (notSortedStorage_.getItemsCount() + shrinkBatchSize - 1) / shrinkBatchSize;
218  auto notSortedKeysStream = getNotSortedKeys();
219  std::vector<std::shared_ptr<io::TemporaryFile>> tempFilesLock;
220  std::vector<KeyIndexStorage> sortedBatches;
221  sortedBatches.reserve(batchesCount + 1);
222  tempFilesLock.reserve(batchesCount + 1);
223 
224  KeyIndexStorage exportedKeys
225  = exportKeys(shrinkBatchSize, tempSortedIndexFilename);
226  tempFilesLock.push_back(exportedKeys.shareStorageFile());
227 
228  sortedBatches.push_back(std::move(exportedKeys));
229  IndexT sortedStorageSize = sortedStorage_.getItemsCount();
230  for (std::size_t batchI = 0; batchI < batchesCount; ++batchI) {
231  const std::string batchFileName = shrinkFilenamePrefix + "-batch-" + std::to_string(batchI);
232  std::vector<KeyIndex> notSortedKeys = notSortedKeysStream.collectWith(
233  [sortedStorageSize](ValueIgnorer &&svi, IndexT index) {
234  return KeyIndex{std::move(svi.key), index + sortedStorageSize};
235  },
236  shrinkBatchSize
237  );
238  KeyIndexStorage sortedBatch(
239  notSortedKeys.begin(),
240  notSortedKeys.end(),
241  false,
242  batchFileName,
243  getFileManager(),
244  [](const KeyIndex &a, const KeyIndex &b) { return a.key < b.key; },
245  [](const KeyIndex &a, const KeyIndex &b) { return a.key == b.key; }
246  );
247  tempFilesLock.push_back(sortedBatch.shareStorageFile());
248  assert(sortedBatch.getItemsCount() > 0 && "Batch file can not be empty");
249  sortedBatches.push_back(std::move(sortedBatch));
250  }
251 
252  KeyIndexStorage updatedIndex(
253  sortedBatches,
254  newIndexFileName,
255  getFileManager(),
256  [](const KeyIndex &a, const KeyIndex &b) { return a.key < b.key; },
257  [](const KeyIndex &a, const KeyIndex &b) { return a.key == b.key; },
258  shrinkBatchSize
259  );
260 
261  resetWith(KeyValueShrinkableStorage(
262  *this,
263  updatedIndex,
264  shrinkFilenamePrefix + "-new-not-sorted",
265  shrinkFilenamePrefix + "-new-sorted",
266  shrinkBatchSize
267  ));
268 
269  KeyIndexStorage actualExportedKeys
270  = exportKeys(shrinkBatchSize, shrinkFilenamePrefix + "-actual-index");
271  tempFilesLock.push_back(actualExportedKeys.shareStorageFile());
272  updatedIndex.resetWith(std::move(actualExportedKeys));
273  return updatedIndex;
274  }

284  KeyIndexStorage exportKeys(IndexT batchSize, const std::string &keysFilename) {
285  KeyIndexStorage sortedIndex(0, keysFilename, getFileManager());
286  auto it = sortedStorage_.template getCustomDataIterator<ValueIgnorer>();
287  auto keys = it.collectWith([](ValueIgnorer &&kvi, IndexT ind) {
288  return KeyIndex{std::move(kvi.key), ind};
289  }, batchSize);

294  return KeyIndex{std::move(kvi.key), ind};
295  }, batchSize);
296  }

298  }
