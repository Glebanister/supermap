233  [[nodiscard]] KeyIndexStorage shrink(IndexT shrinkBatchSize, const std::string &newIndexFileName) {
234 
235  const std::string shrinkFilenamePrefix = "shrink";
236  const std::string tempSortedIndexFilename = shrinkFilenamePrefix + "-sorted-keys";
237 
238  std::size_t batchesCount = (notSortedStorage_.getItemsCount() + shrinkBatchSize - 1) / shrinkBatchSize;
239  auto notSortedKeysStream = getNotSortedKeys();
240  std::vector<std::shared_ptr<io::TemporaryFile>> tempFilesLock;
241  std::vector<KeyIndexStorage> sortedBatches;
242  sortedBatches.reserve(batchesCount + 1);
243  tempFilesLock.reserve(batchesCount + 1);
244 
245  KeyIndexStorage exportedKeys = exportKeys(shrinkBatchSize, tempSortedIndexFilename);
246  tempFilesLock.push_back(exportedKeys.shareStorageFile());
247 
248  sortedBatches.push_back(std::move(exportedKeys));
249  IndexT sortedStorageSize = sortedStorage_.getItemsCount();
250  for (std::size_t batchI = 0; batchI < batchesCount; ++batchI) {
251  const std::string batchFileName = shrinkFilenamePrefix + "-batch-" + std::to_string(batchI);
252  std::vector<KeyIndex> notSortedKeys = notSortedKeysStream.collectWith(
253  [sortedStorageSize](ValueIgnorer &&svi, IndexT index) {
254  return KeyIndex{std::move(svi.key), index + sortedStorageSize};
255  },
256  shrinkBatchSize
257  );
258  KeyIndexStorage sortedBatch(
259  notSortedKeys.begin(),
260  notSortedKeys.end(),
261  false,
262  batchFileName,
263  getFileManager(),
264  [](const KeyIndex &a, const KeyIndex &b) { return a.key < b.key; },
265  [](const KeyIndex &a, const KeyIndex &b) { return a.key == b.key; },
266  innerRegisterSupplier_
267  );
268  tempFilesLock.push_back(sortedBatch.shareStorageFile());
269  assert(sortedBatch.getItemsCount() > 0 && "Batch file can not be empty");
270  sortedBatches.push_back(std::move(sortedBatch));
271  }

304  KeyIndexStorage exportKeys(IndexT batchSize, const std::string &keysFilename) {
305  KeyIndexStorage sortedIndex(keysFilename, getFileManager(), innerRegisterSupplier_);
306  auto it = sortedStorage_.template getCustomDataIterator<ValueIgnorer>();
307  auto keys = it.collectWith([](ValueIgnorer &&kvi, IndexT ind) {
308  return KeyIndex{std::move(kvi.key), ind};
309  }, batchSize);
310  sortedIndex.getRegister().reserve(sortedStorage_.getItemsCount());
311  while (!keys.empty()) {
312  sortedIndex.appendAll(keys.cbegin(), keys.cend());
313  keys = it.collectWith([](ValueIgnorer &&kvi, IndexT ind) {
314  return KeyIndex{std::move(kvi.key), ind};
315  }, batchSize);
