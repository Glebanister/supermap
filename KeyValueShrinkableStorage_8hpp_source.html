217  [[nodiscard]] KeyIndexStorage shrink(IndexT shrinkBatchSize, const std::string &newIndexFileName) {
218 
219  const std::string shrinkFilenamePrefix = "shrink";
220  const std::string tempSortedIndexFilename = shrinkFilenamePrefix + "-sorted-keys";
221 
222  std::size_t batchesCount = (notSortedStorage_.getItemsCount() + shrinkBatchSize - 1) / shrinkBatchSize;
223  auto notSortedKeysStream = getNotSortedKeys();
224  std::vector<std::shared_ptr<io::TemporaryFile>> tempFilesLock;
225  std::vector<KeyIndexStorage> sortedBatches;
226  sortedBatches.reserve(batchesCount + 1);
227  tempFilesLock.reserve(batchesCount + 1);
228 
229  KeyIndexStorage exportedKeys
230  = exportKeys(shrinkBatchSize, tempSortedIndexFilename);
231  tempFilesLock.push_back(exportedKeys.shareStorageFile());
232 
233  sortedBatches.push_back(std::move(exportedKeys));
234  IndexT sortedStorageSize = sortedStorage_.getItemsCount();
235  for (std::size_t batchI = 0; batchI < batchesCount; ++batchI) {
236  const std::string batchFileName = shrinkFilenamePrefix + "-batch-" + std::to_string(batchI);
237  std::vector<KeyIndex> notSortedKeys = notSortedKeysStream.collectWith(
238  [sortedStorageSize](ValueIgnorer &&svi, IndexT index) {
239  return KeyIndex{std::move(svi.key), index + sortedStorageSize};
240  },
241  shrinkBatchSize
242  );
243  KeyIndexStorage sortedBatch(
244  notSortedKeys.begin(),
245  notSortedKeys.end(),
246  false,
247  batchFileName,
248  getFileManager(),
249  [](const KeyIndex &a, const KeyIndex &b) { return a.key < b.key; },
250  [](const KeyIndex &a, const KeyIndex &b) { return a.key == b.key; }
251  );
252  tempFilesLock.push_back(sortedBatch.shareStorageFile());
253  assert(sortedBatch.getItemsCount() > 0 && "Batch file can not be empty");
254  sortedBatches.push_back(std::move(sortedBatch));
255  }
256 
257  KeyIndexStorage updatedIndex(
258  sortedBatches,
259  newIndexFileName,
260  getFileManager(),
261  shrinkBatchSize
262  );
263 
264  resetWith(KeyValueShrinkableStorage(
265  *this,
266  updatedIndex,
267  shrinkFilenamePrefix + "-new-not-sorted",
268  shrinkFilenamePrefix + "-new-sorted",
269  shrinkBatchSize
270  ));
271 
272  KeyIndexStorage actualExportedKeys
273  = exportKeys(shrinkBatchSize, shrinkFilenamePrefix + "-actual-index");
274  tempFilesLock.push_back(actualExportedKeys.shareStorageFile());
275  updatedIndex.resetWith(std::move(actualExportedKeys));
276  return updatedIndex;
